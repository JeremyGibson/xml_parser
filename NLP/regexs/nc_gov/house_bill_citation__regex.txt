### REGEX
(?x)		# set free spacing mode to make REGEX readable.
\b 		# word boundary.
(
(?i)House\s{1}Bill(?-i)\s{1}|
H|
HB\s{0,1}|
H\.\s{0,1}B\.\s{1}
) 		# "House Bill " (case-insensitive) OR "H" OR "HB" (with optional trailing 
		# space)  OR "H.B. " (with optional whitespace after 1st period).
[0-9]{1,4}	# 1 to 4 digits.
\b 		# word boundary.
