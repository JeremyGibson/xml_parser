### REGEX
(?x)		# set free spacing mode to make REGEX readable.
\b 		# word boundary.
(
Executive\s{1}Order\s{1}|
Executive\s{1}Order\s{1}No\.\s{1}|
EO\s{0,1}|
E\.\s{0,1}O\.\s{1}
) 		# "Executive Order " OR "Executive Order No. " OR "EO" (with optional
		# trailing whitespace) OR "E.O. " (with optional whitespace after 1st period).
[0-9]{1,3}	# 1 to 3 digits.
\b 		# word boundary.
